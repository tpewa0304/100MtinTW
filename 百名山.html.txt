        let visitedIds = JSON.parse(localStorage.getItem('visited_100_peaks') || '[]');

        function updateUI() {
            const count = visitedIds.length;
            const total = mountains.length;
            const percent = (count / total) * 100;
            document.getElementById('progress-fill').style.width = percent + '%';
            document.getElementById('progress-text').innerText = `${count} / ${total}`;
        }

        function togglePeak(id) {
            const index = visitedIds.indexOf(id);
            if (index > -1) {
                visitedIds.splice(index, 1);
            } else {
                visitedIds.push(id);
            }
            localStorage.setItem('visited_100_peaks', JSON.stringify(visitedIds));
            location.reload(); // 重新整理以反映地圖顏色變化
        }

        function clearData() {
            if(confirm("確定要清除所有攀登紀錄嗎？")) {
                localStorage.removeItem('visited_100_peaks');
                location.reload();
            }
        }

        const colors = { "百岳": "#d63031", "小百岳": "#00b894", "中級山": "#0984e3", "離島": "#6c5ce7" };

        mountains.forEach(m => {
            const isVisited = visitedIds.includes(m.id);
            const mainColor = isVisited ? "#f1c40f" : (colors[m.type] || "#333");

            // 圓點標記
            const circle = L.circleMarker([m.lat, m.lng], {
                radius: 7, fillColor: mainColor, color: "#fff", weight: 2, fillOpacity: 0.9
            }).addTo(map);

            // 彈窗
            const popupHtml = `
                <div class="popup-card">
                    <h3>${m.id} ${m.name}</h3>
                    <p>海拔分類：${m.type}</p>
                    <label>
                        <input type="checkbox" ${isVisited ? 'checked' : ''} onchange="togglePeak('${m.id}')"> 
                        <b>已登頂 (Mark as Visited)</b>
                    </label>
                </div>
            `;
            circle.bindPopup(popupHtml);

            // 文字標籤
            const labelClass = isVisited ? 'mountain-label visited-label' : 'mountain-label';
            const labelIcon = L.divIcon({
                className: 'label-container',
                html: `<div class="${labelClass}">${m.id} ${m.name}</div>`,
                iconSize: [80, 20], iconAnchor: [40, -12]
            });
            L.marker([m.lat, m.lng], { icon: labelIcon, interactive: false }).addTo(map);
        });

        updateUI();
